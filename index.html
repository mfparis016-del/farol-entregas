String(val).trim() : ''; }); pedidos.push(obj); } return resposta({ pedidos: pedidos, total: pedidos.length, atualizadoEm: Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'dd/MM/yyyy HH:mm:ss') }); } catch (err) { return resposta({erro: err.message}); } } // ------------------------------------------------------------ // POST — atualiza o STATUS de um pedido pelo número do pedido // Body JSON: { pedido: "12345", status: "Entregue" } // ------------------------------------------------------------ function doPost(e) { try { var body = JSON.parse(e.postData.contents); var numeroPedido = String(body.pedido || '').trim(); var novoStatus = String(body.status || '').trim(); if (!numeroPedido || !novoStatus) { return resposta({erro: 'Campos "pedido" e "status" são obrigatórios.'}); } var statusValidos = ['Em rota', 'Aguardando embarque', 'Entregue', 'Problema']; if (statusValidos.indexOf(novoStatus) < 0) { return resposta({erro: 'Status inválido. Use: ' + statusValidos.join(', ')}); } var ss = SpreadsheetApp.openById(SHEET_ID); var aba = ss.getSheetByName(ABA_NOME); if (!aba) return resposta({erro: 'Aba não encontrada.'}); var dados = aba.getDataRange().getValues(); var cabecalho = dados[0].map(function(h) { return String(h).trim().toUpperCase(); }); var colPedido = cabecalho.indexOf('PEDIDO'); var colStatus = cabecalho.indexOf('STATUS'); if (colPedido < 0 || colStatus < 0) { return resposta({erro: 'Colunas PEDIDO ou STATUS não encontradas.'}); } var atualizados = 0; for (var i = 1; i < dados.length; i++) { if (String(dados[i][colPedido]).trim() === numeroPedido) { aba.getRange(i + 1, colStatus + 1).setValue(novoStatus); atualizados++; } } if (atualizados === 0) { return resposta({erro: 'Pedido ' + numeroPedido + ' não encontrado.'}); } return resposta({ok: true, pedido: numeroPedido, novoStatus: novoStatus, linhasAtualizadas: atualizados}); } catch (err) { return resposta({erro: err.message}); } } // ------------------------------------------------------------ // Helper — retorna ContentService com CORS liberado // ------------------------------------------------------------ function resposta(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ------------------------------------------------------------ // Teste rápido — rode esta função manualmente para testar // ------------------------------------------------------------ function testar() { var resultado = doGet(null); Logger.log(resultado.getContent()); }
